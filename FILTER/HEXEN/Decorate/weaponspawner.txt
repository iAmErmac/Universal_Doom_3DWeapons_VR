{
	Health 15
	Radius 15
	Height 32
	Mass 1000
	+SOLID
	+SHOOTABLE
	+NOBLOOD
	+DROPOFF
	+SLIDESONWALLS
	+PUSHABLE
	+TELESTOMP
	+CANPASS
	+NOICEDEATH

	States
	{
	Spawn:
		ZBAR A -1
		Stop

		ZBAR B 2 A_PlaySound("FlechetteExplode")

		ZBAR D 2 A_NoBlocking
		ZBAR EFG 2
		ZBAR H 100
	FadeOut:
		ZBAR H 5 A_FadeOut
		Loop
	}
